263  double
264  integrateTime(IntegrationSchemeType IntegrationScheme,
265  double maxTimeStep = std::numeric_limits<double>::max()) {
266  if (timeStepRatio >= 0.5) {
267  lsMessage::getInstance()
268  .addWarning("Integration time step ratio should be smaller than 0.5. "
269  "Advection might fail!")
270  .print();
271  }
272 
273  auto &topDomain = levelSets.back()->getDomain();
274  auto &grid = levelSets.back()->getGrid();
275 
276  std::vector<std::vector<std::pair<T, T>>> totalTempRates;
277  totalTempRates.resize((levelSets.back())->getNumberOfSegments());
278 
279 #pragma omp parallel num_threads((levelSets.back())->getNumberOfSegments())
280  {
281  int p = 0;
282 #ifdef _OPENMP
283  p = omp_get_thread_num();
284 #endif
285 
286  hrleVectorType<hrleIndexType, D> startVector =
287  (p == 0) ? grid.getMinGridPoint()
288  : topDomain.getSegmentation()[p - 1];
289 
290  hrleVectorType<hrleIndexType, D> endVector =
291  (p != static_cast<int>(topDomain.getNumberOfSegments() - 1))
292  ? topDomain.getSegmentation()[p]
293  : grid.incrementIndices(grid.getMaxGridPoint());
294 
295  double tempMaxTimeStep = maxTimeStep;
296  auto &tempRates = totalTempRates[p];
297  tempRates.reserve(topDomain.getNumberOfPoints() /
298  double((levelSets.back())->getNumberOfSegments()));
299 
300  // an iterator for each level set
301  std::vector<hrleSparseIterator<typename lsDomain<T, D>::DomainType>>
302  iterators;
303  for (auto it = levelSets.begin(); it != levelSets.end(); ++it) {
304  iterators.push_back(
305  hrleSparseIterator<typename lsDomain<T, D>::DomainType>(
306  (*it)->getDomain()));
307  }
308 
309  IntegrationSchemeType scheme(IntegrationScheme);
310 
311  for (hrleSparseIterator<typename lsDomain<T, D>::DomainType> it(
312  topDomain, startVector);
313  it.getStartIndices() < endVector; ++it) {
314 
315  if (!it.isDefined() || std::abs(it.getValue()) > 0.5)
316  continue;
317 
318  const T value = it.getValue();
319  double maxStepTime = 0;
320  double cfl = timeStepRatio;
321 
322  for (int currentLevelSetId = levelSets.size() - 1;
323  currentLevelSetId >= 0; --currentLevelSetId) {
324 
325  T velocity = 0;
326 
327  // check if there is any other levelset at the same point:
328  // if yes, take the velocity of the lowest levelset
329  for (unsigned lowerLevelSetId = 0; lowerLevelSetId < levelSets.size();
330  ++lowerLevelSetId) {
331  // put iterator to same position as the top levelset
332  iterators[lowerLevelSetId].goToIndicesSequential(
333  it.getStartIndices());
334 
335  // if the lower surface is actually outside, i.e. its LS value is
336  // lower or equal
337  if (iterators[lowerLevelSetId].getValue() <= value + 1e-9) {
338  velocity =
339  scheme(it.getStartIndices(), velocities, lowerLevelSetId);
340  break;
341  }
342  }
343 
344  T valueBelow;
345  // get value of material below (before in levelSets list)
346  if (currentLevelSetId > 0) {
347  iterators[currentLevelSetId - 1].goToIndicesSequential(
348  it.getStartIndices());
349  valueBelow = iterators[currentLevelSetId - 1].getValue();
350  } else {
351  valueBelow = std::numeric_limits<T>::max();
352  }
353 
354  // if velocity is positive, set maximum time step possible without
355  // violating the cfl condition
356  if (velocity > 0.) {
357  maxStepTime += cfl / velocity;
358  tempRates.push_back(
359  std::make_pair(velocity, -std::numeric_limits<T>::max()));
360  break;
361  // if velocity is 0, maximum time step is infinite
362  } else if (velocity == 0.) {
363  maxStepTime = std::numeric_limits<T>::max();
364  tempRates.push_back(
365  std::make_pair(velocity, std::numeric_limits<T>::max()));
366  break;
367  // if the velocity is negative apply the velocity for as long as
368  // possible without infringing on material below
369  } else {
370  T difference = std::abs(valueBelow - value);
371 
372  if (difference >= cfl) {
373  maxStepTime -= cfl / velocity;
374  tempRates.push_back(
375  std::make_pair(velocity, std::numeric_limits<T>::max()));
376  break;
377  } else {
378  maxStepTime -= difference / velocity;
379  // the second part of the pair indicates how far we can move in
380  // this time step until the end of the material is reached
381  tempRates.push_back(std::make_pair(velocity, valueBelow));
382  cfl -= difference;
383  }
384  }
385  }
386 
387  if (maxStepTime < tempMaxTimeStep)
388  tempMaxTimeStep = maxStepTime;
389  }
390 
391 #pragma omp critical
392  {
393  // If scheme is STENCIL_LOCAL_LAX_FRIEDRICHS the time step is reduced
394  // depending on the dissipation coefficients For all remaining schemes
395  // this function is empty.
396  reduceTimeStepHamiltonJacobi(scheme, tempMaxTimeStep);
397 
398  // set global timestep maximum
399  if (tempMaxTimeStep < maxTimeStep)
400  maxTimeStep = tempMaxTimeStep;
401  }
402  }
403 
404  // reduce to one layer thickness and apply new values directly to the
405  // domain segments --> DO NOT CHANGE SEGMENTATION HERE (true parameter)
406  lsReduce<T, D>(*levelSets.back()).apply(1, true);
407 
408 #pragma omp parallel num_threads((levelSets.back())->getNumberOfSegments())
409  {
410  int p = 0;
411 #ifdef _OPENMP
412  p = omp_get_thread_num();
413 #endif
414 
415  typename std::vector<std::pair<T, T>>::const_iterator itRS =
416  totalTempRates[p].begin();
417  auto &segment = topDomain.getDomainSegment(p);
418  unsigned maxId = segment.getNumberOfPoints();
419  for (unsigned localId = 0; localId < maxId; ++localId) {
420  T &value = segment.definedValues[localId];
421  double time = maxTimeStep;
422 
423  // if there is a change in materials during one time step, deduct the
424  // time taken to advect up to the end of the top material and set the LS
425  // value to the one below
426  while (std::abs(itRS->second - value) < std::abs(time * itRS->first)) {
427  time -= std::abs((itRS->second - value) / itRS->first);
428  value = itRS->second;
429  ++itRS;
430  }
431 
432  // now deduct the velocity times the time step we take
433  value -= time * itRS->first;
434 
435  // this
436  // is run when two materials are close but the velocity is too slow to
437  // actually reach the second material, to get rid of the extra entry in
438  // the TempRatesStop
439  while (std::abs(itRS->second) != std::numeric_limits<T>::max())
440  ++itRS;
441 
442  // advance the TempStopRates iterator by one
443  ++itRS;
444  }
445  }
446 
447  return maxTimeStep;
448  }
